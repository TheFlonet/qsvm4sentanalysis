    \begin{columns}
        \begin{column}{0.4\textwidth}

            Quadratic Programming Problem
        \end{column}
        \begin{column}{0.1\textwidth}
        \end{column}
        \begin{column}{0.4\textwidth}
            \centering
            QUBO Problem
        \end{column}
    \end{columns}
    \vspace{0.5cm}
    \begin{columns}
        \begin{column}{0.4\textwidth}
            \begin{equation*}
                \begin{aligned}
                    \min_{x} \quad & \frac{1}{2}x^TQx + c^Tx\\
                    \textrm{s.t.} \quad & Ax \preceq b\\
                    &x_i\in \mathbb{R}, \forall i    \\
                \end{aligned}
            \end{equation*}        
        \end{column}
        \begin{column}{0.1\textwidth}
            \LARGE
            $\Longrightarrow$
        \end{column}
        \begin{column}{0.4\textwidth}
            \begin{equation*}
                \begin{aligned}
                    \min_{x} \quad & x^TQx\\
                    &x_i\in\{0, 1\}, \forall i    \\
                \end{aligned}
            \end{equation*}  
        \end{column}
    \end{columns}

\end{frame}

\begin{frame}
    \frametitle{QUBO Formulation}

    The QPU process directly problems in QUBO form:

    \begin{equation*}
        \begin{aligned}
            \min_{x} \quad & x^TQx\\
            &x_i\in\{0, 1\}, \forall i    \\
        \end{aligned}
    \end{equation*}  

    Example with 4 binary variables:

    \begin{center}         
        $\begin{bmatrix}
          x_1 & x_2 & x_3 & x_4
        \end{bmatrix}$
        $\begin{bmatrix}
          -3 & 5 & 1 & -7 \\
          0 & 1 & 0 & 8 \\
          0 & 0 & 9 & -4 \\
          0 & 0 & 0 & -2 
        \end{bmatrix}$
        $\begin{bmatrix}
          x_1 \\ x_2 \\ x_3 \\ x_4
        \end{bmatrix}$      
    \end{center}

\end{frame}
